def myHarrisCornerDetector(img,k):
  f_dim = 7  # dimension of filter
  img = np.pad(img, pad_width=f_dim//2, mode='constant', constant_values=0)
  # coefficients due to euclidian distance (7 x 7 neighbour)
  l, w = img.shape
  x_, y_ = np.meshgrid(np.linspace(-1,1,f_dim), np.linspace(-1,1,f_dim))
  d = np.sqrt(x_*x_+y_*y_)
  sigma_w, mu_w = 0.7, 0.0  # gaussian weights for structure tensor
  sigma_I, mu_I = 0.42, 0.0  # gaussian before structure tensor
  gauss_sp = np.exp(-((d-mu_w)**2 / (2.0*sigma_w**2))) # gaussian distribution
  gauss_sp = gauss_sp/np.sum(gauss_sp)
  gauss_I = np.exp(-((d-mu_I)**2 / (2.0*sigma_I**2)))
  gauss_I = gauss_I/np.sum(gauss_I)
  pad = f_dim//2
  output = np.zeros((l-2*pad, w-2*pad), dtype="float32")
  for y in np.arange(pad, l - pad):
    for x in np.arange(pad, w - pad):
      # region of interest from the input image in each iteration
      roi = img[y - pad:y + pad + 1, x - pad:x + pad + 1]
      # intensity difference with this pixel's in the region (7 x 7 neighbour
      mul = (roi * gauss_I).sum() #gaussian before computing structure tensor
      output[y - pad, x - pad] = mul
  out_rshift = np.zeros(output.shape)
  out_ushift = np.zeros(output.shape)
  out_rshift[:,0] = output[:,0]
  out_ushift[0,:] = output[0,:]
  out_rshift[:,1:] = output[:,:-1]
  out_ushift[1:,:] = output[:-1,:]
  x_deriv = output - out_rshift
  y_deriv = output - out_ushift
  Ix2 = np.array(x_deriv)*np.array(x_deriv)
  Iy2 = np.array(y_deriv)*np.array(y_deriv)
  Ixy = np.array(x_deriv)*np.array(y_deriv)
  pad = f_dim//2
  outputix2 = np.zeros((l-2*pad, w-2*pad), dtype="float32")
  outputiy2 = np.zeros((l-2*pad, w-2*pad), dtype="float32")
  outputixy = np.zeros((l-2*pad, w-2*pad), dtype="float32")
  Ix2 = np.pad(Ix2, pad_width=f_dim//2, mode='constant', constant_values=0)#Ix^2
  Iy2 = np.pad(Iy2, pad_width=f_dim//2, mode='constant', constant_values=0)#Iy^2
  Ixy = np.pad(Ixy, pad_width=f_dim//2, mode='constant', constant_values=0)#Ixy
  for y in np.arange(pad, l - pad):
    for x in np.arange(pad, w - pad):
      # region of interest from the input image in each iteration
      roiix2 = Ix2[y - pad:y + pad + 1, x - pad:x + pad + 1]
      roiiy2 = Iy2[y - pad:y + pad + 1, x - pad:x + pad + 1]
      roiixy = Ixy[y - pad:y + pad + 1, x - pad:x + pad + 1]
      # intensity difference with this pixel's in the region (7 x 7 neighbour)
      #ints = roi - img[y][x]
      # point by point multiplication of space coefficients with intensity
      # coefficients after computing later from the gaussian distribution
      # result gauss_ints corresponds to the bilateral filter coefficients
      # multiply the filter values pointwise in region of interest and add
      mulix2= (roiix2 * gauss_sp).sum()
      muliy2= (roiiy2 * gauss_sp).sum()
      mulixy= (roiixy * gauss_sp).sum()
      outputix2[y - pad, x - pad] = mulix2
      outputiy2[y - pad, x - pad] = muliy2
      outputixy[y - pad, x - pad] = mulixy
  R = np.array(outputix2)*np.array(outputiy2)-np.array(outputixy)*np.array(outputixy) - k*(np.array(outputix2+outputiy2)*np.array(outputix2+outputiy2))
  sumix2iy2 = outputix2+outputiy2
  delta = np.array(sumix2iy2)*np.array(sumix2iy2) - 4*(np.array(outputix2)*np.array(outputiy2)-np.array(outputixy)*np.array(outputixy))
  lamda1 = (sumix2iy2 + np.sqrt(delta))/2
  lamda2 = (sumix2iy2 - np.sqrt(delta))/2
  return R,x_deriv,y_deriv,lamda1,lamda2 # float [0, 1]
